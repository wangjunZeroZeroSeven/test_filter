
set(CMAKE_CXX_STANDARD 17)
cmake_minimum_required(VERSION 3.10)
project(test_filter VERSION 0.1.0)

include(CTest)
enable_testing()

add_executable(test_filter main.cpp)

set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)


# conan config
include(${CMAKE_BINARY_DIR}/conanbuildinfo.cmake)
if (APPLE)
    conan_basic_setup(TARGETS         # Allow to target_link packets as CONAN_PKG::packet
                    NO_OUTPUT_DIRS  # Do not adjust the build output directories.
                    KEEP_RPATHS
                    )
else ()
    conan_basic_setup(TARGETS         # Allow to target_link packets as CONAN_PKG::packet
                    NO_OUTPUT_DIRS  # Do not adjust the build output directories.
                    )
endif()
# Following packages will be found by using FindXXX.cmake generated by conan
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_BINARY_DIR})
find_package(spdlog 1.8.2 REQUIRED)
target_link_libraries(test_filter PUBLIC spdlog::spdlog)
target_include_directories(test_filter PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} ${spdlog_INCLUDE_DIRS})
# conan config
